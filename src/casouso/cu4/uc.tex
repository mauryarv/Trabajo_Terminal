\begin{UCtrayectoria}
	
	\UCpaso[\UCactor] Da clic en la opción "Consultar horario".
	
	\UCpaso[\UCsist] Obtiene la información del horario seleccionado.
	
	\UCpaso[\UCsist] Muestra la información del resultado.	
	
	\UCpaso[\UCactor] Solcita terminar la consulta presionando el botón \cdtButton{Regresar}. 
	
	\UCpaso[\UCsist] Muestra la pantalla anterior.
	
\end{UCtrayectoria}
