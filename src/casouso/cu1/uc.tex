%Trayectoria principal ----------------------------------------------------------------------------

\begin{UCtrayectoria}
	
	\UCpaso [\UCactor]	Abre la aplicacion sin haber iniciado sesión con anterioridad. 
	\UCpaso [\UCsist]Muestra la pantalla de inicio de sesión con dos espaciós ingresar su correo o nombre de usuario y contraseña.
	\UCpaso [\UCactor]Debera escribir correo o nombre de usuario registrados con anterioridad y su respectiva contraseña y dar a \cdtButton{Iniciar Sesión}.\refTray{A} \refTray{B}
	\UCpaso [\UCsist]Muestra la interfaz principal.
\end{UCtrayectoria}


% Trayectorias alternativas -----------------------------------------------------------------------

% Trayectoria A
\begin{UCtrayectoriaA}{A}{Sin cuenta previamente registrada}
	\UCpaso [\UCsist]Muestra mensaje: \cdtRef{MSG}{Esta cuenta no esta registrada}.
	\UCpaso [\UCactor]	Selecciona \cdtButton{Registrarse}
	\UCpaso Va al paso 1 del CU 2.
\end{UCtrayectoriaA}

% Trayectoria B
\begin{UCtrayectoriaA}{B}{Contraseña incorrecta}
	\UCpaso [\UCsist]Muestra mensaje: \cdtRef{MSG}{La contraseña es incorrecta}.
	\UCpaso [\UCactor]	Vuelve a ingresar la contraseña o dar a \cdtButton{Recuperar contraseña}.
	\UCpaso Regresa al paso 3 de la trayectoria principal.
\end{UCtrayectoriaA}
