\begin{UCtrayectoria}
	
	\UCpaso[\UCactor] Da clic en la opción "Modificar el provedor".
	
	\UCpaso[\UCsist] Obtiene la información del provedor agregado o seleccionado.
	
	\UCpaso[\UCsist] Muestra la información del resultado.	
	
	\UCpaso[\UCactor] Solcita terminar de modificar provedor o productos, presionando el botón \cdtButton{Regresar}. 
	
	\UCpaso[\UCsist] Muestra la pantalla con un mensaje producto o provedor modificado con exito.
	
\end{UCtrayectoria}
