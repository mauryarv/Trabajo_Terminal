\begin{UCtrayectoria}
	
	\UCpaso[\UCactor] Da clic en la opción \cdtButton{Proveedores} de la pantalla \cdtIdRef{Menu principal}{Index}.

	\UCpaso[\UCsist] Muestra pantalla \cdtIdRef{Menu Proveedores}{Index}.

	\UCpaso[\UCactor] Da clic en la opción \cdtButton{Gestionar proveedor}.

	\UCpaso[\UCsist] Obtiene todas las relaciones de los proveedores registrados en la bd. \refTray{A}

	\UCpaso[\UCsist] Muestra la pantalla \cdtIdRef{---- Gestionar proveedor}{Index}.

	\UCpaso[\UCactor] Da clic en el icono engrane de la sección 'ajustes' del proveedor al que desee modificar.

	\UCpaso[\UCsist] Obtiene información del proveedor seleccionado y la muestra en la pantalla \cdtIdRef{---- Modificar proveedor}{Index}. \refTray{B}

	\UCpaso[\UCactor] Realiza las modificaciones deseadas en los campos disponibles para modificación. \refTray{C} 

	\UCpaso[\UCactor] Da clic en el boton \cdtButton{Guardar cambios}. 

	\UCpaso[\UCsist] Realiza cambios y actualiza la bd. \refTray{D}

	\UCpaso[\UCsist] Muestra el mensaje 1 'El proveedor fue actualizado con éxito' en la pantalla \cdtIdRef{-------- Alerta éxito al actualizar}{Index}.

	\UCpaso[\UCactor] Da clic sobre 'ok'.

	\UCpaso[\UCsist] Muestra la pantalla \cdtIdRef{Menu proveedores}{Index}.
	
\end{UCtrayectoria}
